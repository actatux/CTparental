
## mise a jour os + install git
su -
urpmi.update -a 
urpmi --update --auto-select 
urpmi git
exit

## désinstaler l'ancienne versions.
su -
CTparental -u
exit

## téléchargement du scripte CTparental pour mageia
git clone https://github.com/marsat/CTparental.git
cd CTparental/

## install et configuration de CTparental
su -
cp -f CTparental /usr/bin
cp -f CTparental-bl-infos /usr/bin
mkdir /etc/CTparental
mkdir /usr/share/CTparental
cp -rf www /usr/share/CTparental
cp -rf confDansgouardian /usr/share/CTparental
cp -f locale/fr_FR/LC_MESSAGES/ctparental.mo		/usr/share/locale/fr_FR/LC_MESSAGES/
cd mageia
CTparental -i
exit
